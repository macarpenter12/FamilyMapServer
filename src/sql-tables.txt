CREATE TABLE user_table(
    username    String PRIMARY KEY,
    password    String NOT NULL,
    email       String,
    firstName   String NOT NULL,
    lastName    String,
    gender      String
);
CREATE TABLE person_table(
    personID    String PRIMARY KEY,
    username    String,
    firstName   String,
    lastName    String,
    gender      String,
    fatherID    String,
    motherID    String,
    spouseID    String
);
CREATE TABLE event_table(
    eventID     String PRIMARY KEY,
    username    String,
    personID    String,
    latitude    FLOAT,
    longitude   FLOAT,
    country     String,
    city        String,
    type        String,
    year        INTEGER
);
CREATE TABLE authtoken_table(
    authToken   String PRIMARY KEY,
    username    String NOT NULL
);
